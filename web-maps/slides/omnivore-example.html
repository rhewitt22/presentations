Map tiles by Stamen Design, CC BY 3.0 — Map data © OpenStreetMap 

  var map = L.map('map').setView([33.775133, -84.381808], 9);
  var neighborhoods = omnivore.kml('../data/neighborhoods.kml')
    .on('ready', function() {
      map.fitBounds(neighborhoods.getBounds());
    }).addTo(map);
